131  Face * addFace(VertexInputIterator vbegin, VertexInputIterator vend)
132  {
133  // Check for errors and compute normal
134  size_t num_verts = 0;
135  Vector3 v[3];
136  for (VertexInputIterator vi = vbegin; vi != vend; ++vi, ++num_verts)
137  {
138  debugAssertM(*vi, getNameStr() + ": Null vertex pointer specified for new face");
139  if (num_verts < 3) v[num_verts] = (*vi)->getPosition();
140  }
141 
142  if (num_verts < 3)
143  {
144  DGP_WARNING << getName() << ": Skipping face -- too few vertices (" << num_verts << ')';
145  return NULL;
146  }
147 
148  // Create the (initially empty) face
149  faces.push_back(Face());
150  Face * face = &(*faces.rbegin());
151 
152  // Add the loop of vertices to the face
153  VertexInputIterator next = vbegin;
154  for (VertexInputIterator vi = next++ ; vi != vend; ++vi, ++next)
155  {
156  if (next == vend) next = vbegin;
157 
158  face->addVertex(*vi);
159  (*vi)->addFace(face, false); // we'll update the normals later
160 
161  Edge * edge = (*vi)->getEdgeTo(*next);
162  if (!edge)
163  {
164  edges.push_back(Edge(*vi, *next));
165  edge = &(*edges.rbegin());
166 
167  (*vi)->addEdge(edge);
168  (*next)->addEdge(edge);
169  }
170 
171  edge->addFace(face);
172  face->addEdge(edge);
173  }
174 
175  // Update the face and vertex normals;
176  face->updateNormal();
177  for (typename Face::VertexIterator fvi = face->verticesBegin(); fvi != face->verticesEnd(); ++fvi)
178  (*fvi)->addFaceNormal(face->getNormal()); // weight by face area?
179 
180  return face;
181  }
