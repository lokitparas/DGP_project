54 
55  // If winding order is: vector to first vertex, ray, vector to second vertex, then intersects
56  Vector3 c0 = v0.cross(u);
57  Vector3 c1 = u.cross(v1);
58  if (c0.dot(c1) > 0) // intersects, now check forward or reverse
59  {

61  //
62  // The point p' on line v + t * e closest to point p is v + t0 * e, where t0 = e.dot(p - v) / e.dot(e)
63  // (see www.geometrictools.com/Documentation/DistancePointLine.pdf).
64  //

68  Vector3 u0 = v0 + t0 * edge;
