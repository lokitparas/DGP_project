12 Mesh::mergeEdges(Edge * e0, Edge * e1)
13 {
14  if (!e0) return e1;
15  if (!e1) return e0;
16 
17  alwaysAssertM(e0->isCoincidentTo(*e1), std::string(getName()) + ": Edges to merge must have the same endpoints");
18 
19  // Transfer faces from e1 to e0
20  for (Edge::FaceIterator fi = e1->facesBegin(); fi != e1->facesEnd(); ++fi)
21  {
22  if (!e0->hasIncidentFace(*fi))
23  e0->addFace(*fi);
24 
25  (*fi)->replaceEdge(e1, e0); // in case the face references both e1 and e0, we need to do the replacement even if the above
26  // 'if' condition is false
27  }
28 
29  Edge * edges_to_remove[2] = { e1, NULL };
30  if (e0->numFaces() <= 0)
31  edges_to_remove[1] = e0;
32 
33  // Need to cache these before e0 is potentially removed
34  Vertex * u = e0->getEndpoint(0);
35  Vertex * v = e0->getEndpoint(1);
36 
37  for (int i = 0; i < 2; ++i)
38  {
39  if (!edges_to_remove[i])
40  continue;
41 
42  edges_to_remove[i]->getEndpoint(0)->removeEdge(edges_to_remove[i]);
43  edges_to_remove[i]->getEndpoint(1)->removeEdge(edges_to_remove[i]);
44 
45  for (EdgeIterator ej = edges.begin(); ej != edges.end(); ++ej)
46  if (&(*ej) == edges_to_remove[i])
47  {
48  edges.erase(ej);
49  break;
50  }
51  }
52 
53  Vertex * vertices_to_remove[2] = { NULL, NULL };
54  if (u->numFaces() <= 0 && u->numEdges() <= 0) vertices_to_remove[0] = u;
55  if (v->numFaces() <= 0 && v->numEdges() <= 0) vertices_to_remove[1] = v;
56 
57  for (int i = 0; i < 2; ++i)
58  {
59  if (!vertices_to_remove[i])
60  continue;
61 
62  for (VertexIterator vj = vertices.begin(); vj != vertices.end(); ++vj)
63  if (&(*vj) == vertices_to_remove[i])
64  {
65  vertices.erase(vj);
66  break;
67  }
68  }
69 
70  return (edges_to_remove[1] == e0 ? NULL : e0);

73 MeshVertex *
74 Mesh::collapseEdge(Edge * edge)
75 {
76  if (!edge)
77  return NULL;
78 
79  Vertex * u = edge->getEndpoint(0);
80  Vertex * v = edge->getEndpoint(1);
81 
82  if (u == v)
83  {
84  DGP_CONSOLE << getName() << ": Can't collapse edge that is self loop";
85  return NULL;
86  }
87 
88  // Check if u is a repeated vertex in any face. If so, preferentially remove it (one copy at a time)
89  bool stop = false;
90  for (FaceConstIterator fi = faces.begin(); fi != faces.end(); ++fi)
91  {
92  int num_occurrences = 0;
93  for (MeshFace::VertexConstIterator vi = fi->verticesBegin(); vi != fi->verticesEnd(); ++vi)
94  {
95  if (*vi == u)
96  {
97  num_occurrences++;
98  if (num_occurrences >= 2) // u is the one to remove
99  {
100  std::swap(u, v);
101  stop = true;
102  break;
103  }
104  }
105  }
106 
107  if (stop)
108  break;
109  }
110 
111  // Remove edge and v from adjacent faces
112  std::vector<Face *> edge_faces(edge->facesBegin(), edge->facesEnd());
113  for (size_t i = 0; i < edge_faces.size(); ++i)
114  {
115  Face * face = edge_faces[i];
116 
117  // The edge and vertex have to be removed in linked fashion from the loops around the face. This is tricky.
118  while (true)
119  {
120  MeshFace::VertexIterator vi = face->verticesBegin();
121  bool found = false;
122  for (MeshFace::EdgeIterator ei = face->edgesBegin(); ei != face->edgesEnd(); ++ei, ++vi)
123  {
124  if (*ei != edge)
125  continue;
126 
127  face->edges.erase(ei);
128  if (*vi == v)
129  face->vertices.erase(vi);
130  else
131  {
132  ++vi;
133  bool last = false;
134  if (vi == face->verticesEnd()) { vi = face->verticesBegin(); last = true; }
135  if (*vi == v)
136  {
137  face->vertices.erase(vi);
138 
139  if (last) // we need to do a shift
140  {
141  MeshVertex * b = face->vertices.back();
142  face->vertices.pop_back();
143  face->vertices.push_front(b);
144  }
145  }
146  else
147  {
148  DGP_ERROR << getName() << ": Vertex not found in expected location";
149  return NULL;
150  }
151  }
152 
153  found = true;
154  break;
155  }
156 
157  if (!found)
158  break;
159  }
160 
161  // Wrap up with the easy stuff
162  edge->removeFace(face);
163  v->removeFace(face);
164  }
165 
166  assert(edge->numFaces() == 0);
167 
168  // Faces adjacent to the edge do not reference it, or v, any more. 'edge' is isolated except for its references to u and v.
169  // The mesh is in an inconsistent state since the face does not reference v whereas the next edge around the face does.
170 
171  for (Vertex::EdgeIterator vei = v->edgesBegin(); vei != v->edgesEnd(); ++vei)
172  {
173  (*vei)->replaceVertex(v, u); // this also makes edge = (u, u)
174 
175  if (!u->hasIncidentEdge(*vei))
176  u->addEdge(*vei);
177  }
178 
179  v->edges.clear();
180 
181  assert(v->numEdges() == 0);
182 
183  // No edges reference v any more. The mesh is in an inconsistent state since faces incident to v but not containing 'edge' do
184  // not contain u yet.
185 
186  for (Vertex::FaceIterator vfi = v->facesBegin(); vfi != v->facesEnd(); ++vfi)
187  {
188  (*vfi)->replaceVertex(v, u); // no-op if we had removed v from this face above
189 
190  if (!u->hasIncidentFace(*vfi))
191  u->addFace(*vfi);
192  }
193 
194  v->faces.clear();
195 
196  assert(v->numFaces() == 0);
197 
198  // No faces reference v any more. The mesh is in a consistent state.
199 
200  for (EdgeIterator ei = edges.begin(); ei != edges.end(); ++ei)
201  if (&(*ei) == edge)
202  {
203  edges.erase(ei);
204  break;
205  }
206 
207  u->removeEdge(edge);
208 
209  // No more edge. The mesh is in a consistent state
210 
211  for (VertexIterator vi = vertices.begin(); vi != vertices.end(); ++vi)
212  if (&(*vi) == v)
213  {
214  vertices.erase(vi);
215  break;
216  }
217 
218  // No more v. The mesh is in a consistent state.
219 
220  for (size_t i = 0; i < edge_faces.size(); ++i)
221  {
222  Face * face = edge_faces[i];
223  if (face->numVertices() >= 3)
224  continue;
225 
226  removeFace(face);
227  }
228 
229  // All faces shrunk to zero by the edge collapse have been removed.
230 
231  bool found = true;
232  while (found)
233  {
234  found = false;
235 
236  for (Vertex::EdgeIterator vei = u->edgesBegin(); vei != u->edgesEnd(); ++vei)
237  {
238  for (Vertex::EdgeIterator vej = u->edgesBegin(); vej != vei; ++vej)
239  if ((*vei)->isCoincidentTo(**vej))
240  {
241  mergeEdges(*vei, *vej);
242  found = true;
243  break;
244  }
245 
246  if (found)
247  break;
248  }
249  }
250 
251  // All double edges have been collapsed to single edges (this can happen either because faces were shrunk to zero, or because
252  // the mesh has genus > 0).
253 
254  return u;
255 }

258 Mesh::draw(Graphics::RenderSystem & render_system, bool draw_edges, bool use_vertex_data, bool send_colors) const
259 {
260  // Three separate passes over the faces is probably faster than using Primitive::POLYGON for each face
261 
262  if (draw_edges)
263  {
264  render_system.pushShapeFlags();
265  render_system.setPolygonOffset(true, 1);
266  }
267 
268  // First try to render as much stuff using triangles as possible
269  render_system.beginPrimitive(Graphics::RenderSystem::Primitive::TRIANGLES);
270  for (FaceConstIterator fi = facesBegin(); fi != facesEnd(); ++fi)
271  if (fi->isTriangle()) drawFace(*fi, render_system, use_vertex_data, send_colors);
272  render_system.endPrimitive();
273 
274  // Now render all quads
275  render_system.beginPrimitive(Graphics::RenderSystem::Primitive::QUADS);
276  for (FaceConstIterator fi = facesBegin(); fi != facesEnd(); ++fi)
277  if (fi->isQuad()) drawFace(*fi, render_system, use_vertex_data, send_colors);
278  render_system.endPrimitive();
279 
280  // Finish off with all larger polygons
281  for (FaceConstIterator fi = facesBegin(); fi != facesEnd(); ++fi)
282  if (fi->numEdges() > 4)
283  {
284  render_system.beginPrimitive(Graphics::RenderSystem::Primitive::POLYGON);
285  drawFace(*fi, render_system, use_vertex_data, send_colors);
286  render_system.endPrimitive();
287  }
288 
289  if (draw_edges)
290  render_system.popShapeFlags();
291 
292  if (draw_edges)
293  {
294  render_system.pushShader();
295  render_system.pushColorFlags();
296 
297  render_system.setShader(NULL);
298  render_system.setColor(ColorRGBA(0.2, 0.3, 0.7, 1)); // set default edge color
299 
300  render_system.beginPrimitive(Graphics::RenderSystem::Primitive::LINES);
301  for (EdgeConstIterator ei = edgesBegin(); ei != edgesEnd(); ++ei)
302  {
303  render_system.sendVertex(ei->getEndpoint(0)->getPosition());
304  render_system.sendVertex(ei->getEndpoint(1)->getPosition());
305  }
306  render_system.endPrimitive();
307 
308  render_system.popColorFlags();
309  render_system.popShader();
310  }

314 Mesh::loadOFF(std::string const & path)
315 {
316  std::ifstream in(path.c_str());
317  if (!in)
318  {
319  DGP_ERROR << "Could not open '" << path << "' for reading";
320  return false;
321  }
322 
323  clear();
324 
325  std::string magic;
326  if (!(in >> magic) || magic != "OFF")
327  {
328  DGP_ERROR << "Header string OFF not found at beginning of file '" << path << '\'';
329  return false;
330  }
331 
332  long nv, nf, ne;
333  if (!(in >> nv >> nf >> ne))
334  {
335  DGP_ERROR << "Could not read element counts from OFF file '" << path << '\'';
336  return false;
337  }
338 
339  if (nv < 0 || nf < 0 || ne < 0)
340  {
341  DGP_ERROR << "Negative element count in OFF file '" << path << '\'';
342  return false;
343  }
344 
345  std::vector<Vertex *> indexed_vertices;
346  Vector3 p;
347  for (long i = 0; i < nv; ++i)
348  {
349  if (!(in >> p[0] >> p[1] >> p[2]))
350  {
351  DGP_ERROR << "Could not read vertex " << indexed_vertices.size() << " from '" << path << '\'';
352  return false;
353  }
354 
355  Vertex * v = addVertex(p);
356  if (!v)
357  return false;
358 
359  indexed_vertices.push_back(v);
360  }
361 
362  std::vector<Vertex *> face_vertices;
363  long num_face_vertices, vertex_index;
364  for (long i = 0; i < nf; ++i)
365  {
366  if (!(in >> num_face_vertices) || num_face_vertices < 0)
367  {
368  DGP_ERROR << "Could not read valid vertex count of face " << faces.size() << " from '" << path << '\'';
369  return false;
370  }
371 
372  face_vertices.resize(num_face_vertices);
373  for (size_t j = 0; j < face_vertices.size(); ++j)
374  {
375  if (!(in >> vertex_index))
376  {
377  DGP_ERROR << "Could not read vertex " << j << " of face " << faces.size() << " from '" << path << '\'';
378  return false;
379  }
380 
381  if (vertex_index < 0 || vertex_index >= (long)vertices.size())
382  {
383  DGP_ERROR << "Out-of-bounds index " << vertex_index << " of vertex " << j << " of face " << faces.size() << " from '"
384  << path << '\'';
385  return false;
386  }
387 
388  face_vertices[j] = indexed_vertices[(size_t)vertex_index];
389  }
390 
391  addFace(face_vertices.begin(), face_vertices.end()); // ok if this fails, just skip the face with a warning
392  }
393 
394  setName(FilePath::objectName(path));
395 
396  return true;
397 }
